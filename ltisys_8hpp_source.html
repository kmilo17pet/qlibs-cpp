162 template<size_t numberOfDelays>
163 class transportDelay : public ITransportDelay {
164 private:
165 real_t buf[ numberOfDelays + 1 ];
166 tdl dl;
167 public:
168 virtual ~transportDelay() {}
174 transportDelay( const real_t initValue = 0.0_re )
175 {
176 static_assert( numberOfDelays >= 1 , "Delay taps should be greater than 0" );
177 dl.setup( buf, initValue);
178 }
179
185 real_t delay( const real_t xInput ) noexcept override
186 {
187 dl.insertSample( xInput );
188 return dl.getOldest();
189 }
190
196 real_t operator()( const real_t xInput ) noexcept override
197 {
198 return delay( xInput );
199 }
200
201 constexpr size_t getNumberOfDelays() const noexcept {
202 return numberOfDelays;
203 }
